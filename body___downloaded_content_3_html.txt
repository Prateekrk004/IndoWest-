<script>
        // Header scroll effect
        window.addEventListener('scroll', () => {
            const header = document.querySelector('header');
            if (window.scrollY > 50) {
                header.classList.add('shadow-sm');
            } else {
                header.classList.remove('shadow-sm');
            }
        });

        // Filter functionality (Simulation)
        const checkboxes = document.querySelectorAll('input[type="checkbox"]');
        checkboxes.forEach(checkbox => {
            checkbox.addEventListener('change', (e) => {
                const label = e.target.nextElementSibling;
                if(e.target.checked) {
                    label.classList.add('text-primary', 'font-semibold');
                    label.classList.remove('text-on-surface');
                } else {
                    label.classList.remove('text-primary', 'font-semibold');
                    label.classList.add('text-on-surface');
                }
            });
        });

        // Add to favorite toggle
        const favButtons = document.querySelectorAll('.product-card button .material-symbols-outlined');
        favButtons.forEach(btn => {
            btn.addEventListener('click', (e) => {
                e.stopPropagation();
                const icon = e.target;
                if(icon.style.fontVariationSettings.includes("'FILL' 1")) {
                    icon.style.fontVariationSettings = "'FILL' 0, 'wght' 300";
                    icon.classList.remove('text-primary');
                } else {
                    icon.style.fontVariationSettings = "'FILL' 1, 'wght' 300";
                    icon.classList.add('text-primary');
                }
            });
        });
    </script>
